```c
 #include <stdio.h>
 #define NUMBER  80
 
 int main (void)
{
	int i,j;
	int num;
	int tensu[NUMBER];
	int bunpu[11] = {0};
	
	printf("请输入学生人数：");
	do{
		scanf("%d",&num);
		if(num < 1||num > NUMBER)
			printf("\a请输入1~%d的数：",NUMBER); 
	} while(num < 1||num > NUMBER) ;
	
	printf("请输入%d人的分数。\n",num);
	
	for(i = 0;i < num;i++){
		printf("%2d号：",i + 1);
		do{
			scanf("%d",&tensu[i]);
			if (tensu[i] < 0||tensu[i] > 100)
				printf("\a请输入1~100的数：");
				 
		}while(tensu[i] < 0||tensu[i] > 100);
		bunpu[tensu[i] / 10]++;    //得出分数分布 
		
	}
	puts("\n---分布图---");
	int max = bunpu[1];
	for(i = 2;i <=11;i++){
		if (bunpu[i] > max)
			max = bunpu[i];
	}       //得出最多的频数
	for(; max > 0;max--){ //输出星星 
		 for (j = 0; j < 11; j++){
            if (bunpu[j] >= max)
                printf("  *   ");
            else
                printf("      ");
        }
        putchar('\n');
	}
	printf("--------------------------------------------------------------------");
	putchar('\n');
	for(i = 0;i <= 10;i++)
		printf("  %d  ",i * 10);
		
	return 0; 
 }
```

```c
 #include <stdio.h>
 
 int main (void)
{
	int i,j;
	int tensu1[4][3] = {{91, 63, 78},{89, 34, 53},{32, 54, 34}};
	int tensu2[4][3] = {{97, 67, 82},{97, 56, 21},{85, 46, 35}};
	int sum[4][3];//总分 
	
	for(i = 0;i < 4;i++){
		for(j = 0;j < 3;j++)
			sum[i][j] = tensu1[i][j] + tensu2[i][j];
	}
	puts("第一次考试的分数");
	for(i = 0;i < 4;i++){
		for(j = 0;j < 3;j++)
			printf("%4d",tensu1[i][j]);
		putchar('\n');
	}
	puts("第二次考试的分数");
	for(i = 0;i < 4;i++){
		for(j = 0;j < 3;j++)
			printf("%4d",tensu2[i][j]);
		putchar('\n');
	}
	puts("总分");
	for(i = 0;i < 4;i++){
		for(j = 0;j < 3;j++)
			printf("%4d",sum[i][j]);
		putchar('\n');
	}
		
	return 0; 
 }
```
